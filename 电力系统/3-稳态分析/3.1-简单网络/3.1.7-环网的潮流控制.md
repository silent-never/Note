\left. \begin{aligned} S_{a1} &= \frac{(\overset{*}{Z}_{12} + \overset{*}{Z}_{b2})S_1 + \overset{*}{Z}_{b2}S_2}{\overset{*}{Z}_{a1} + \overset{*}{Z}_{12} + \overset{*}{Z}_{b2}} + \frac{(\dot{U}_a - \dot{U}_b)U_{\mathrm{N}}}{\overset{*}{Z}_{a1} + \overset{*}{Z}_{12} + \overset{*}{Z}_{b2}} = S_{a1.\mathrm{LD}} + S_{\mathrm{cir}} \\ S_{b2} &= \frac{\overset{*}{Z}_{a1}S_1 + (\overset{*}{Z}_{a1} + \overset{*}{Z}_{12})S_2}{\overset{*}{Z}_{a1} + \overset{*}{Z}_{12} + \overset{*}{Z}_{b2}} - \frac{(\dot{U}_a - \dot{U}_b)U_{\mathrm{N}}}{\overset{*}{Z}_{a1} + \overset{*}{Z}_{12} + \overset{*}{Z}_{b2}} = S_{b2.\mathrm{LD}} - S_{\mathrm{cir}} \end{aligned} \right\}
