\end{bmatrix} = 
\begin{bmatrix} 
Y_{11} & Y_{12} \\[4pt] 
Y_{21} & Y_{22} 
\end{bmatrix} 
\begin{bmatrix} 
\dot{U}_1 \\ 
\dot{U}_2 
\end{bmatrix} 
$$

$$
\begin{bmatrix} 
\dot U_1 \\

\begin{bmatrix}

\end{bmatrix} 
\begin{bmatrix} 
\dot{U}_2 \\ 
\dot{I}_2
\end{bmatrix} 
$$
节点导纳矩阵——二端口网络：
$$
-\dot I_2=Y_{21}\dot U_1+Y_{22}\dot U_2
$$
$$
\dot U_1=\frac{-\dot I_2 -Y_{22}\dot U_2}{Y_{21}}
$$
$$
\begin{aligned} 
\dot I_1 &=-Y_{11}\frac{\dot I_2 +Y_{22}\dot U_2}{Y_{21}}+Y_{22}\dot U_2\\
&=(Y_{22}-\frac{Y_{11}Y_{22}}{Y_{21}})\dot U_2-\frac{Y_{11}}{Y_{21}}\dot I_2
\end{aligned}
$$
$$
\begin{bmatrix} 
\dot U_1 \\ 
\dot I_1 
\end{bmatrix} = 
\begin{bmatrix} 
-\frac{1}{Y_{21}} &-\frac{Y_{22}}{Y_{21}} \\[4pt] 
Y_{22}-\frac{Y_{11}Y_{22}}{Y_{21}}&-\frac{Y_{11}}{Y_{21}}
\end{bmatrix} 
\begin{bmatrix} 
\dot{U}_2 \\ 
\dot{I}_2
\end{bmatrix} 
$$
二端口网络——节点导纳矩阵：
$$
\dot U_1=A\dot U_2+B\dot I_2
$$
$$
\dot I_2=\frac{\dot U_1-A\dot U_2}{B}
$$
$$
\begin{aligned} 
\dot I_1&=C\dot U_2+D\frac{\dot U_1-A\dot U_2}{B}\\
&=-\frac{D}{B}\dot U_1+(C-\frac{AD}{B})\dot U_2
\end{aligned}
$$
$$
\begin{bmatrix} 
\dot{I}_1 \\ 
-\dot{I}_2 
\end{bmatrix} = 
\begin{bmatrix} 
-\frac{D}{B} & C-\frac{AD}{B} \\[4pt] 
-\frac{1}{B} & \frac{A}{B} 
\end{bmatrix} 
\begin{bmatrix} 
\dot{U}_1 \\ 
\dot{U}_2 
\end{bmatrix}
