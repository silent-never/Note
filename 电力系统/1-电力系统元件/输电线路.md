$$
\left. \begin{aligned} 
\frac{\mathrm{d}\dot{U}}{\mathrm{d}x} &= -(r_0 + \mathrm{j}\omega l_0)\dot{I} \\ 
\frac{\mathrm{d}\dot{I}}{\mathrm{d}x} &= -(g_0 + \mathrm{j}\omega c_0)\dot{U} 
\end{aligned} \right\}
$$
$$
\frac{\mathrm{d}^2\dot{U}}{\mathrm{d}x^2}= (r_0 + \mathrm{j}\omega l_0)(g_0 + \mathrm{j}\omega c_0)\dot{U} =\gamma^2\dot{U}
$$
$$
U=A_U \mathrm{ch}(\gamma x)+B_U \mathrm{sh}(\gamma x)
$$
同理有：
$$
I=A_I \mathrm{ch}(\gamma x)+B_I \mathrm{sh}(\gamma x)
$$
带入一阶导数关系，
边界条件：
$$
\left. \begin{aligned} 
\dot{U}|_{x=0} &=U_1 \\ 
\dot{I}|_{x=0} &=I_1
\end{aligned} \right\}
$$
