$$
\left. \begin{aligned} 
\frac{\mathrm{d}\dot{U}}{\mathrm{d}x} &= -(r_0 + \mathrm{j}\omega l_0)\dot{I} \\ 
\frac{\mathrm{d}\dot{I}}{\mathrm{d}x} &= -(g_0 + \mathrm{j}\omega c_0)\dot{U} 
\end{aligned} \right\}
$$
$$
\frac{\mathrm{d}^2\dot{U}}{\mathrm{d}x^2}= (r_0 + \mathrm{j}\omega l_0)(g_0 + \mathrm{j}\omega c_0)\dot{U} =\tilde{\gamma}^2\dot{U}
$$
$$
\dot U=\dot A_U \mathrm{ch}(\gamma x)+\dot B_U \mathrm{sh}(\gamma x)
$$
同理有：
$$
\dot I=\dot A_I \mathrm{ch}(\gamma x)+\dot B_I \mathrm{sh}(\gamma x)
$$
边界条件：
$$
\left. \begin{aligned} 
\dot{U}|_{x=0} &=\dot U_1 \\ 
\dot{I}|_{x=0} &=\dot I_1
\end{aligned} \right\}
$$
有
$$
\left. \begin{alignedat}{2}
\dot U&=\dot U_1 \mathrm{ch}(\gamma x)&&+\dot B_U \mathrm{sh}(\gamma x) \\ 
\dot I&=\dot I_1 \mathrm{ch}(\gamma x)&&+\dot B_I  \mathrm{sh}(\gamma x)
\end{alignedat} \right\}
$$
带入一阶导数关系:
$$
\left. \begin{alignedat}{2}
\tilde{\gamma} \dot U_1 &= -(r_0 + \mathrm{j}\omega l_0)\dot B_I \\ 
\tilde{\gamma} \dot B_U &= -(r_0 + \mathrm{j}\omega l_0)\dot I_1
\end{alignedat} \right\}
$$
取
$$
\tilde{Z_c}=\sqrt\frac{r_0 + \mathrm{j}\omega l_0}{g_0 + \mathrm{j}\omega c_0}=\frac{}{\tilde{\gamma}}
